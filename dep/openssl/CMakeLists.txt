if(EXISTS ${CMAKE_BINARY_DIR}/../openssl)
  set(OPENSSL_DIR ${CMAKE_BINARY_DIR}/../openssl CACHE PATH "")
else()
  set(OPENSSL_DIR ${CMAKE_CURRENT_SOURCE_DIR}/openssl CACHE PATH "")
endif()

  set(OPENSSL_ROOT_DIR ${OPENSSL_DIR}/inc32 ${OPENSSL_DIR}/out32dll PARENT_SCOPE)
else()
  set(OPENSSL_ROOT_DIR ${OPENSSL_DIR} PARENT_SCOPE)
endif()
